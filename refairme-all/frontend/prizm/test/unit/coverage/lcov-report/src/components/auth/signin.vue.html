<template lang="pug">
.container(style="margin-top:20px;margin-bottom:20px;")
  .col-12.col-md-4.offset-md-4(style="background:rgba(173,216,230,0.25); padding-bottom:20px; padding-top:20px; border-radius: 5px;")
    .col-12
      .panel.panel-default
        .panel-heading(style="text-align: center") Sign In
        .panel-body
          form(action="" v-on:submit.prevent="login()" method="post")
            small.help-block(style="color: red") {{error}}
            .form-group
              label(for="email") Email
              input.form-control(type="email" name="email" id="email" placeholder="you@domain.com" v-model="aemail")
            .form-group
              label(for="password") Password
              input.form-control(type="password" name="password" id="password" v-model="apassword")
            button.btn.btn-outline-primary(style="width: 100%" type="submit") Sign In
</template>
<script>
import helpy from '../../helpers.js'
export default {
  name: 'signin',
  mounted(){
    this.error = this.$route.params.info
  },
  data: function() {
    return {
      apassword: '',
      aemail: '',
      error: ''
    }
  },
  computed: {
    isAuthenticated: function() {
      return this.$store.getters.isAuthenticated
    }
  },
  methods: {
    login: function() {
      const locmail = this.$data.aemail;
      const locpass = this.$data.apassword;
 
      this.$store.dispatch('signin', {
        locmail,
        locpass
      }).then(ret => {
        console.log(ret);
        var data = ret.data;
        if(data.state === 'error'){
          this.error = data.message
        } else {
          this.$router.push('/profilebuild')
        }
      });
    }
  }
}
</script>
