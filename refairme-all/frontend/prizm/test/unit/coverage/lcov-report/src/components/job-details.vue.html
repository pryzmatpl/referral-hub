<template lang="pug">
div
  .card
    .card-body
      .alert(v-bind:class="[applyResponse.state == 'success' ? alertClasses.success : alertClasses.error]", v-if='applyResponse.message', role='alert')
        | {{applyResponse.message}}
      h1
        | {{job.jobtitle}}
        button.btn.btn-success.float-right.col-3(@click='apply()')
          | Apply
          font-awesome-icon(v-if='loading', :icon='cogIcon', spin='')
      h4.text-muted {{job.required_fund[0]}} - {{job.required_fund[1]}} PLN
      small.badge.badge-pill.badge-light REMOTE = {{job.required_remote}}
      small.badge.badge-pill.badge-light RELOCATION = {{job.required_relocation}}
  .card.mt-3
    .card-body
      h4 Description
      .row
        .col-6.text-muted(v-html="job.description")
        .col-6
          lineChartKw(style='width: 100%', chart-id='kwchart', :height='250', :chart-data='patterndatakw')
</template>
<script>
import Vue from 'vue'
import VueChartJs from 'vue-chartjs'
import {
  mixins,
  Bar,
  Line,
  Radar,
  HorizontalBar
} from 'vue-chartjs'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import {
  faCog
} from '@fortawesome/fontawesome-free-solid'
 
var lineChartKw = {
  extends: VueChartJs.HorizontalBar,
  mixins: [mixins.reactiveProp],
  props: ['chartData'],
  data(){
    return {
      options:{
        responsive: true,
        maintainAspectRatio: false,
        legend: {
          display: false
        }
      }
    }
  },
  mounted() {
    if (this.chartData) {
      this.renderChart(this.chartData, this.options)
    }
  }
};
 
export default {
  mounted() {
    var id = this.$route.params.id;
    this.$store.state.backend
    .get('api/job/' + id, {
      headers: {
        'Access-Control-Allow-Origin': '*'
      },
      crossDomain: true,
    })
    .then(ret => {
      this.$data.job = ret.data.jobdata;
      console.log(ret.data.jobdata)
      //this.$data.patterndatakw.datasets[0].data = ret.data.weights;
      this.matchJobs();
      console.log(ret.data.weights)
      this.$data.patterndatakw = {
        labels: ['Backend', 'Full Stack', 'Mobile/Embedded', 'Testing', 'Frontend', 'Dev Ops', 'Business Intelligence', 'IT Trainee', 'Project Management', 'Support', 'UX Designer', 'Business Analyst', 'Other'],
        datasets: [{
          label: 'Refair.me Profile',
          backgroundColor: '#a84979',
          data: ret.data.weights
        }]
      }
    });
  },
  computed: {
    cogIcon() {
      //https://www.npmjs.com/package/@fortawesome/vue-fontawesome
      return faCog
    }
  },
  data() {
    return {
      job: {
        jobtitle: 'Title',
        description: 'Desc',
        required_remote: true,
        required_relocation: true,
        required_fund: '10000'
      },
      patterndatakw: {
        labels: ['Backend', 'Full Stack', 'Mobile/Embedded', 'Testing', 'Frontend', 'Dev Ops', 'Business Intelligence', 'IT Trainee', 'Project Management', 'Support', 'UX Designer', 'Business Analyst', 'Other'],
        datasets: [{
          label: 'Refair.me Profile ',
          backgroundColor: '#a84979',
          data: []
        }]
      },
      alertClasses: {
        success: 'alert-success',
        error: 'alert-danger'
      },
      loading: false,
      applyResponse: {}
    }
  },
  methods: {
    apply: function() {
      this.loading = true;
      var id = this.$route.params.id;
      const self = this;
      this.$store.state.backend.post('/api/apply', {
        jobid: id,
        email: this.$store.state.dehashedData.EMAIL
      }).then(function(ret) {
        self.loading = false;
        console.log('applied')
        console.log(ret.data)
        self.applyResponse = ret.data
      })
    },
    matchJobs: function() {
      var id = this.$route.params.id;
      console.log("match jobs")
      const self = this;
      this.$store.state.backend.get('/matchjob/' + id, {
        weights: self.patterndatakw.datasets[0].data
      }).then(function(ret) {
        console.log('matchJobs = success')
        console.log(ret)
      })
    }
  },
  components:{
    FontAwesomeIcon,
    lineChartKw
  }
}
</script>
