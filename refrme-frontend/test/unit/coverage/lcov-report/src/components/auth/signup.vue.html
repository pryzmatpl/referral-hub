<template lang="pug">
.container(style='margin-top:20px;margin-bottom:20px;')
  .col-12.col-md-4.offset-md-4(style='background:rgba(173,216,230,0.25); padding-bottom:20px; padding-top:20px; border-radius: 5px;')
    .col-12
      .panel.panel-default
        .panel-heading(style='text-align: center') Sign Up
        .panel-body
          form(action='', v-on:submit.prevent='signUserUp()', method='post')
            small.help-block(style='color: red;') {{error}}
            .form-group
              label(for='email') Email
              input#email.form-control(type='email', name='email', placeholder='you@domain.com', v-model='email')
            .form-group
              label(for='password') Password
              input#password.form-control(type='password', name='password', v-model='password')
            button.btn.btn-outline-primary(style='width: 100%', type='submit') Sign Up
</template>
<script>
import helpy from '../../helpers.js'

export default {
  name: 'signup',
  beforeCreate: function() {
    console.log(this.$route.fullPath);
  },
  data: function() {
    return {
      email: '',
      password: '',
      error: '',
      group: '',
      actionurl: process.env.BACKEND_URL
    }
  },
  methods: {
    signUserUp() {
      var headerRegister = {
        'Content-Type': 'multipart/form-data',
        'Authorization': 'Basic REGISTER',
        'Access-Control-Allow-Origin': '*',
      };
 
      var headerConfirm = {
        'Content-Type': 'multipart/form-data',
        'Authorization': 'Basic CONFIRM',
        'Access-Control-Allow-Origin': '*',
      }
 
      this.$store.state.backend.post('/auth/signup', {
        email: this.email,
        password: this.password,
        chosengroup: this.group,
        headerRegister,
        dehashed: {
          SESSION_AUTH: false,
          SESSION_STATE: 2,
          SESSION_ID: '123ssdsdsd',
          ORIGIN: 'prizm',
          TIMESTAMP: '' //?
        }
      }).then(ret => {
        var data = ret.data;
        if(data.state === 'error'){
          this.error = data.message
        } else {
          this.$router.push({
            name: 'signin',
            params: {
              info: 'Please confirm your account in the email we sent you - you will be able to log in here after confirming your account'
            }
          })
        }
      }).catch(error => {
        //TODO on backend side
        console.log('error')
      });
    }
  }
}
</script>
