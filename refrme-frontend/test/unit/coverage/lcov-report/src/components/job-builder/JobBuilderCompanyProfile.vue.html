<script>
import validation from '../../validation.js'

export default {
  data: function() {
    return {
      companyName: '',
      companyUrl: '',
      contactPerson: '',
      contactEmail: '',
      aboutCompany: ''
    }
  },
  mounted() {
    var self = this;
    $(document).on('change', '.custom-file :file', function() {
      //console.log('file changed');
      $('.custom-file-label').css('color','inherit')
      var input = $(this),
          label = input.val().replace(/\\/g, '/').replace(/.*\//, '');
      input.trigger('fileselect', [label]);
    });
 
    $('.custom-file :file').on('fileselect', function(event, label) {
      var input = $(this).parents('.custom-file').find('label'),
          log = label;
      if (input.length) {
        input[0].innerText = log;
      } else {
        if (log) alert(log);
      }
    });
 
    $("#customFile").change(function() {
      self.readURL(this);
    });
 
    $('form input, textarea').keydown(function(){
      validation.removeError(this)
    })
  },
  methods: {
    readURL: function(input) {
      if (input.files && input.files[0]) {
        var reader = new FileReader();
        reader.onload = function(e) {
          $('#img-upload').attr('src', e.target.result);
        }
        reader.readAsDataURL(input.files[0]);
      }
    },
    validateForm: function(){
      var validated = true;
      var self = this;
 
      $("form#companyProfile label:not(.form-check-label)")
      .each(function(){
        //console.log($(this))
        if(validation.validateField($(this)) == false){
          validated = false
          window.scroll(0,0)
        };
      })
 
      if(!validated){ //TODO change to normal
        this.$emit('job',{
          companyName: this.companyName,
          companyUrl: this.companyUrl,
          contactPerson: this.contactPerson,
          contactEmail: this.contactEmail,
          aboutCompany: this.aboutCompany
        }, 'profile')
      }
 
      //console.log('validated: ' + validated)
      return validated;
    }
  }
}
</script>
