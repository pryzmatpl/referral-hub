<script>
import validation from '../../validation.js'
import {
  mixins,
  //Doughnut,
  PolarArea
} from 'vue-chartjs'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import {
  faTag
} from '@fortawesome/fontawesome-free-solid'
 
var chart = {
  extends: PolarArea,
  mixins: [mixins.reactiveProp],
  props: ['options'],
  mounted() {
    if (this.chartData) {
      this.renderChart(this.chartData, this.options)
    }
  }
};

export default{
  data: function(){
    return {
      whyWorkOnProject: '',
      teamSize: '',
      projectStage: '',
      projectStack: '',
      projectMethodology: [],
      workBreakdown: {
        newFeatures: 20,
        maintenance: 20,
        support: 20,
        documentation: 20,
        meetings: 20
      },
      chartData: null,
      options:{
        responsive: true,
        maintainAspectRatio: false,
        legend: {
          display: false
        },
        layout: {
          padding: 5
        }
      },
      labelColors: ['#10B13A','#EC2E15','#EC8015','#117992','#FFB770']
    }
  },
  mounted: function(){
    this.updateChart(1)
  },
  computed: {
    tagIcon() {
      //https://www.npmjs.com/package/@fortawesome/vue-fontawesome
      return faTag
    }
  },
  watch: {
    'workBreakdown.newFeatures': function(n,o){
      this.updateChart(n)
    },
    'workBreakdown.maintenance': function(n,o){
      this.updateChart(n)
    },
    'workBreakdown.support': function(n,o){
      this.updateChart(n)
    },
    'workBreakdown.documentation': function(n,o){
      this.updateChart(n)
    },
    'workBreakdown.meetings': function(n,o){
      this.updateChart(n)
    }
  },
  methods: {
    updateChart: function(v) {
      if(v != ''){
        this.chartData = {
          labels: [
            'New features',
            'Maintenance / bug fixing',
            'Client support',
            'Document writing',
            'Meetings'
          ],
          datasets: [{
            backgroundColor: this.labelColors,
            data: [
              this.workBreakdown.newFeatures,
              this.workBreakdown.maintenance,
              this.workBreakdown.support,
              this.workBreakdown.documentation,
              this.workBreakdown.meetings
            ]
          }]
        }
      }
    },
    validateForm: function(){
      var validated = true;
      var self = this;

      $("form#aboutProject .form-group label:not(.form-check-label)")
      .each(function(){
        if(validation.validateField($(this)) == false){
          validated = false
          window.scroll(0,0)
        };
      })

      if(!validated){ //TODO change to normal
        this.$emit('job',{
          whyWorkOnProject: this.whyWorkOnProject,
          teamSize: this.teamSize,
          projectStage: this.projectStage,
          projectStack: this.projectStack,
          projectMethodology: this.projectMethodology
        },'aboutProject')
      }
 
      return validated;
    }
  },
  components: {
    chart,
    FontAwesomeIcon
  }
}
</script>
