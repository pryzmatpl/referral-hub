<template>
<div>
  <select class="" data-show-subtext="true" id="selectlocation" name="selectocation" data-live-search="true" style="padding:10px;" @change="setLoc">
    <option :id="location.id" class="selectoptionlocation"
      v-for="location in locations"
      @mouseup="passthis"
      :data-locationid="location.id"
      :data-currency="location.currency"
      :data-locname="location.name" >
      {{location.name}}
    </option>
    <option data-subtext="ALL-LOCATIONS" selected>
      Select location to list available positions
    </option>
  </select>
  <p>Selected:{{namedlocation}}</p>
</div>
</template>
<script>
export default {
  data: function() {
    return {
      namedlocation: '',
      locations: [],
      currency: ' ',
      chosenid: 0
    }
  },
  methods: {
    setLoc: function(event) {
      var actualData = $(event)[0].target.selectedOptions[0];
 
      this.chosenid = $(actualData).data('locationid');
      this.currency = $(actualData).data('currency');
      this.namedlocation = $(actualData).data('locname');

      this.$parent.$emit('setLocs',
        this.chosenid,
        this.namedlocation,
        this.currency);
    },
    passthis: function(event) {
      var lid = $(event.target).data('locationid');
      var anamedlocation = $(event.target).data('locname');
      var acurrency = $(event.target).data('currency');

      this.currency = acurrency;
      this.chosenid = lid;
      this.namedlocation = anamedlocation;

      this.$parent.alocation = this.chosenid;
      this.$parent.currency = this.currency;
 
      this.$parent.$emit('setLoc',
        event,
        this.namedlocation,
        this.currency);
    },
  },
  name: 'refairlocationsoptions',
  mounted: function() {
    this.$http.get('/getlocations', {}).then(ret => {
      this.locations = ret.data; // should work as json comes directly from database
    });
  }
}
</script>
