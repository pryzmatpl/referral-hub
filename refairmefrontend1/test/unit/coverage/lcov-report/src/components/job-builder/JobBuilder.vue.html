<template lang="pug">
  .card
    .card-body
      ul.nav.nav-justified
        li.nav-item(v-for="tab in tabs")
          a.nav-link(
            :class="['.nav-link', {active: currentTab.name === tab.name}]"
            :key="tab.name"
            @click="currentTab = tab"
          ) {{tab.name}}
      keep-alive
        transition(name="fade")
          component.tab.mt-4(
            v-bind:is="currentTabComponent"
            v-on:job="updateJob"
            v-on:add-job="addJob"
          )
</template>
<script>

import JobBuilderCompanyProfile from './JobBuilderCompanyProfile'
import JobBuilderAboutProject from './JobBuilderAboutProject'
import JobBuilderAboutJob from './JobBuilderAboutJob'
import JobBuilderConfirmation from './JobBuilderConfirmation'
 
export default{
  components:{
    JobBuilderCompanyProfile,
    JobBuilderAboutProject,
    JobBuilderAboutJob,
    JobBuilderConfirmation
  },
  data: function(){
    return {
      currentTab: {
        name: 'Create a company profile',
        comp: 'JobBuilderCompanyProfile'
      },
      tabs: [
        {
          name: 'Create a company profile',
          comp: 'JobBuilderCompanyProfile',
        },
        {
          name: 'Describe the project',
          comp: 'JobBuilderAboutProject'
        },
        {
          name:'Present the job',
          comp: 'JobBuilderAboutJob'
        },
        {
          name: 'Submit',
          comp: 'JobBuilderConfirmation'
        }
      ],
      jobBuilder: {}
    }
  },
  computed: {
    currentTabComponent: function () {
      return this.currentTab.comp
    }
  },
  methods: {
    updateJob: function(object,source){
      //console.log(object)
      //console.log(source)
      this.jobBuilder[source] = object
    },
    selectActive: function(event){
      var siblings = $(event.target).parent().siblings().children('a')
      $(event.target).addClass('active')
      siblings.removeClass('active')
    },
    addJob(event) {
      console.log('adding jobs')
 
      this.$store.state.backend
      .post('/api/job/new', this.jobBuilder)
      .then(ret => {
        alert('Sent job and added!');
        console.log(ret);
      });
    },
  }
}
</script>
