<script>
import store from '.../../../prizm-vuex.js'
import refairkeywords from '../keywords.vue'
import slider from 'vue-slider-component'
import validation from '../../validation.js'
 
export default {
  data: function() {
    return {
      keywords: [],
      title: '',
      jobdesc: '',
      location: '',
      contractType: '',
      travel: 20,
      remoteWork: 20,
      currency: '$',
      salary: [3000, 15000],
      requiredexp: 3,
      requiredremote: false,
      requiredrelocation: false,
      relocationPackage: false,
      posterid: this.$store.getters.dehashedData.EMAIL,
      isAuthenticated: this.$store.getters.isAuthenticated
    }
  },
  watch: {
    keywords: function(){
      if(this.keywords.length > 0)
        validation.removeError($('#kwErrMsg'))
    }
  },
  methods: {
    keywordsChange(eventData) {
      this.keywords = eventData
    },
    changeCurrency: function(event){
      var currencies = ['$','€','PLN']
      var index = currencies.indexOf(this.currency) + 1
      if(index == 3) index = 0
      this.currency = currencies[index]
    },
    validateForm: function(){
      var validated = true;
      var self = this;
      $("form#aboutJob label:not(.form-check-label)").each(function(){
        if(validation.validateField($(this)) == false
          || validation.validateKeywords(self.keywords, $('#kwErrMsg')) == false){
          validated = false
          window.scroll(0,0)
        };
      })

      if(!validated){ //TODO change to normal
        this.$emit('job',{
          title: this.title,
          jobdesc: this.jobdesc,
          location: this.location,
          contractType: this.contractType,
          travel: this.travel,
          remoteWork: this.remoteWork,
          salary: this.salary,
          requiredexp: this.requiredexp,
          requiredremote: this.requiredremote,
          requiredrelocation: this.requiredrelocation,
          relocationPackage: this.relocationPackage
        }, 'aboutJob')
      }

      return validated;
    }
  },
  components: {
    refairkeywords,
    slider
  }
}
</script>
