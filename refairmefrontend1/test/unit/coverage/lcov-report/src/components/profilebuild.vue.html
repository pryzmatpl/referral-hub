<template lang="pug">
  div
    .card
      .card-body
        .col-xs-12
          h1.card-title(style='color: #B0AFAB;') Your Profile Builder
        .row
          .col-xs-12.col-md-6
            RefairKeywords#profilekeywords(v-on:keywords='updateProfileKeywords')
          .col-xs-12.col-md-6
            ChartJs(chart-id='kwchart', :height='250', :chart-data='patterndatakw')
        .col-xs-12
          a.btn.btn-info(href='#', @click='saveProfile') Save Profile
    .card.mt-3
      .card-body
        h1(style='color: #B0AFAB;')
          | Profiled Jobs
          font-awesome-icon(v-if='loading', :icon='cogIcon', spin='')
        table.table.table-striped
          thead
            tr.text-muted
              th(scope='col') Id
              th(scope='col') Title
              th(scope='col') Pay
              th(scope='col') Job Profile
          tbody
            tr(v-if='jobs.length == 0')
              td(colspan='4', style='text-align: center') No jobs matched
            tr(v-for='job in jobs', @click='openJobDetails(job.id)')
              th(scope='row') {{job.id}}
              td {{job.title}}
              td {{job.payment[0]}} - {{job.payment[1]}}
              td(style='width: 50%')
                ChartJs(chart-id='', :height='250', :chart-data='populateWeights(job.weights)')
</template>
<script>
import Vue from 'vue'
import {
  mixins,
  HorizontalBar,
} from 'vue-chartjs'
import RefairKeywords from './keywords.vue'
import FontAwesomeIcon from '@fortawesome/vue-fontawesome'
import {
  faCog
} from '@fortawesome/fontawesome-free-solid'

var ChartJs = {
  extends: HorizontalBar,
  mixins: [mixins.reactiveProp],
  props: ['chartData'],
  data(){
    return {
      options:{
        responsive: true,
        maintainAspectRatio: false,
        legend: {
          display: false
        }
      }
    }
  },
  mounted() {
    if (this.chartData) {
      this.renderChart(this.chartData, this.options)
    }
  }
};
 
export default {
  name: 'profile-build',
  mounted: function() {
    var email = this.$store.state.dehashedData.EMAIL;
    this.$store.state.backend.post('/api/user/getprofile/' + email)
    .then(function(response) {
      console.log(response)
    })
  },
  computed: {
    cogIcon() {
      //https://www.npmjs.com/package/@fortawesome/vue-fontawesome
      return faCog
    }
  },
  data() {
    return {
      profilekeywords: [],
      jobs: [],
      loading: false,
      patterndatakw: {
        labels: ['Backend', 'Full Stack', 'Mobile/Embedded', 'Testing', 'Frontend', 'Dev Ops', 'Business Intelligence', 'IT Trainee', 'Project Management', 'Support', 'UX Designer', 'Business Analyst', 'Other'],
        datasets: [{
          label: 'Refair.me Profile ',
          backgroundColor: '#a84979',
          data: [0.05, 0.2, 0.1, 0.5, 0.2, 0.05, 0, 0, 0, 0, 0]
        }]
      }
    }
  },
  methods: {
    populateWeights: function(weights) {
      return {
        labels: ['Backend', 'Full Stack', 'Mobile/Embedded', 'Testing', 'Frontend', 'Dev Ops', 'Business Intelligence', 'IT Trainee', 'Project Management', 'Support', 'UX Designer', 'Business Analyst', 'Other'],
        datasets: [{
          label: 'Refair.me Profile ',
          backgroundColor: '#a84979',
          data: weights
        }]
      }
    },
    updateProfileKeywords: function(value) {
      if(value.length == 0){
        this.jobs = []
      } else {
        this.profilekeywords = value;
        this.loading = true
      }
    },
    saveProfile: function() {
      console.log('Saving profile')
      this.$store.state.backend
      .post('/api/user/storeprofile', {
        params: {
          'email': this.$store.state.dehashedData.EMAIL,
          'weights': this.patterndatakw.datasets[0].data
        }
      })
      .then(function(response) {
        console.log('Profile saved')
      })
    },
    matchProfile: function() {
      console.log("req: matchprofile");
      const self = this;
      this.$store.state.backend
      .get('/matchprofile', {
        crossDomain: true,
        headers: {
          'Access-Control-Allow-Origin': '*',
          'planck': this.$store.state.planck
        },
        params: {
          'passedWeights': JSON.stringify(self.patterndatakw.datasets[0].data)
        }
      })
      .then(function(ret) {
        self.jobs = ret.data
        console.log("res: matchprofile = success");
        console.log(ret.data)
      })
      .catch(function(error) {
        console.error(error);
      })
      .finally(() => this.loading = false);
    },
    openJobDetails: function(jobId) {
      this.$router.push({
        path: `api/job/${jobId}`
      })
    }
  },
  watch: {
    profilekeywords: function(keywords) {
      if(keywords.length != 0){
        const self = this;
        console.log("req: eval");
        // On keywords updated, send it to backend for ai analisys
        this.$store.getters.backend('/eval', {
          crossDomain: true,
          headers: {
            'Access-Control-Allow-Origin': process.env.BACKEND_URL
          },
          params: {
            'eval': keywords.join(),
          }
        }).then(function(ret) {
          console.log("res: eval = success");
          console.log(ret);
          var dataUpd = {
              labels: ['Backend', 'Full Stack', 'Mobile/Embedded', 'Testing', 'Frontend', 'Dev Ops', 'Business Intelligence', 'IT Trainee', 'Project Management', 'Support', 'UX Designer', 'Business Analyst', 'Other'],
              datasets: [{
                label: 'Your Personal Profile ',
                backgroundColor: '#a84979',
                data: ret.data.weightsA.predictions
              }]
            };
          self.patterndatakw = dataUpd;
          self.matchProfile();
        }).catch(function(error) {
          console.error(error);
        });
      }
    }
  },
  components: {
    RefairKeywords,
    FontAwesomeIcon,
    ChartJs
  }
}
</script>
